Modern CSS-Based Enter/Leave Animations:

// CSS Keyframes
@keyframes fade-in {
  from { opacity: 0; }
  to { opacity: 1; }
}

@keyframes slide-in-left {
  from { 
    transform: translateX(-100%); 
    opacity: 0; 
  }
  to { 
    transform: translateX(0); 
    opacity: 1; 
  }
}

.fade-enter { animation: fade-in 300ms ease-in; }
.slide-enter { animation: slide-in-left 300ms ease-in; }

// Template with new animate.enter API

Modern CSS State Transitions:

// CSS State Classes with Transitions
.state-demo-box {
  transition: all 500ms ease-in-out;
  
  &.expanded { height: 200px; width: 300px; }
  &.collapsed { height: 100px; width: 200px; }
  
  &.blue { background-color: #3498db; }
  &.red { background-color: #e74c3c; }
  &.green { background-color: #27ae60; }
  &.purple { background-color: #9b59b6; }
  
  &.rotated { 
    transform: rotate(45deg); 
    transition: transform 300ms ease; 
  }
}

// Template with CSS class bindings
<div 
  class="state-demo-box"
  [class.expanded]="isExpanded"
  [class.collapsed]="!isExpanded"
  [class.blue]="currentColor === 'blue'"
  [class.red]="currentColor === 'red'"
  [class.rotated]="isRotated">
  
  @if (isExpanded) {
    <p>State: Expanded</p>
  } @else {
    <p>State: Collapsed</p>
  }
</div>

}

Modern CSS List Animations:

// CSS Keyframes for List Enter Animation
@keyframes list-enter {
  from {
    opacity: 0;
    transform: translateY(-10px) scale(0.95);
  }
  to {
    opacity: 1;
    transform: translateY(0) scale(1);
  }
}

.list-enter {
  animation: list-enter 200ms ease-out;
}

// Template with new animate.enter API
<div class="animated-list">
  @for (listItem of listItems; track listItem.id) {
    <div class="list-item" animate.enter="list-enter">
      {{ '{' }}{{ '{' }} listItem.text {{ '}' }}{{ '}' }}
    </div>
  } @empty {
    <p>No items in the list</p>
  }
</div>

Modern CSS Keyframe Animations:

// CSS Keyframes (No Angular animations package needed!)
@keyframes bounce {
  0% { transform: translateY(0); }
  30% { transform: translateY(-30px); }
  50% { transform: translateY(0); }
  70% { transform: translateY(-15px); }
  100% { transform: translateY(0); }
}

@keyframes pulse {
  0% { transform: scale(1); }
  50% { transform: scale(1.1); }
  100% { transform: scale(1); }
}

@keyframes wobble {
  0% { transform: rotate(0deg); }
  25% { transform: rotate(-5deg); }
  50% { transform: rotate(5deg); }
  75% { transform: rotate(-3deg); }
  100% { transform: rotate(0deg); }
}

// Animation Classes
.bounce-animation { animation: bounce 600ms ease-out; }
.pulse-animation { animation: pulse 800ms ease-out; }
.wobble-animation { animation: wobble 600ms ease-out; }
